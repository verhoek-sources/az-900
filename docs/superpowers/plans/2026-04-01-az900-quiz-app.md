# AZ-900 Quiz App — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a portable, offline-capable AZ-900 quiz app as two files (`index.html` + `questions.js`) that runs in any browser with no install or server.

**Architecture:** `questions.js` holds a global `QUESTIONS` array. `index.html` contains all HTML structure, CSS, and JavaScript. Three screens (home, quiz, results) are toggled via a `.hidden` CSS class. Score history persists in `localStorage`.

**Tech Stack:** Vanilla HTML5, CSS3, JavaScript (ES6) — zero dependencies, zero build step, works fully offline.

---

| File | Responsibility |
|---|---|
| `questions.js` | Global `QUESTIONS` array — 60 AZ-900 questions with options, answer index, and explanation |
| `index.html` | App shell: HTML structure for 3 screens, all CSS, all JavaScript logic |

Replace with:
```html
  <script>
    // ── Constants ─────────────────────────────────────────────────
    const DOMAIN_BADGE = {
      'Cloud Concepts':                  'badge-cc',
      'Azure Architecture & Services':   'badge-arch',
      'Azure Management & Governance':   'badge-gov'
    };
    const DOMAIN_COLOR = {
      'Cloud Concepts':                  '#0078D4',
      'Azure Architecture & Services':   '#107C10',
      'Azure Management & Governance':   '#8764B8'
    };
    const DOMAIN_WEIGHT = {
      'Cloud Concepts':                  '25–30%',
      'Azure Architecture & Services':   '35–40%',
      'Azure Management & Governance':   '30–35%'
    };
    const HISTORY_KEY    = 'az900-quiz-history';
    const PASS_THRESHOLD = 0.70;

    // ── State ──────────────────────────────────────────────────────
    let shuffled    = [];
    let currentIdx  = 0;
    let userAnswers = [];  // [{ topic, correct }]
    let answered    = false;

    // ── Utilities ──────────────────────────────────────────────────
    function shuffle(arr) {
      const a = [...arr];
      for (let i = a.length - 1; i > 0; i--) {
        const j = Math.floor(Math.random() * (i + 1));
        [a[i], a[j]] = [a[j], a[i]];
      }
      return a;
    }

    function showScreen(id) {
      document.querySelectorAll('.screen').forEach(s => s.classList.add('hidden'));
      document.getElementById(id).classList.remove('hidden');
      window.scrollTo(0, 0);
    }

    // ── localStorage helpers ───────────────────────────────────────
    function saveRun(correct, total) {
      const history = loadHistory();
      history.unshift({
        date:    new Date().toLocaleDateString(),
        correct,
        total,
        percent: Math.round((correct / total) * 100)
      });
      try {
        localStorage.setItem(HISTORY_KEY, JSON.stringify(history.slice(0, 5)));
      } catch (_) { /* localStorage unavailable (e.g. file:// in some browsers) */ }
    }

    function loadHistory() {
      try {
        return JSON.parse(localStorage.getItem(HISTORY_KEY)) || [];
      } catch (_) {
        return [];
      }
    }

    // ── Home Screen ────────────────────────────────────────────────
    function goHome() {
      initHome();
      showScreen('screen-home');
    }

    function initHome() {
      document.getElementById('total-count').textContent = QUESTIONS.length;

      const history     = loadHistory();
      const histSection = document.getElementById('history-section');
      const histList    = document.getElementById('history-list');

      if (history.length === 0) {
        histSection.classList.add('hidden');
        return;
      }

      histSection.classList.remove('hidden');
      histList.innerHTML = history.map(h => `
        <div class="history-item">
          <span class="history-date">${h.date}</span>
          <span class="history-score">${h.correct}/${h.total} &mdash; ${h.percent}%</span>
        </div>
      `).join('');
    }

    // ── Quiz Screen ────────────────────────────────────────────────
    function startQuiz() {
      shuffled    = shuffle(QUESTIONS);
      currentIdx  = 0;
      userAnswers = [];
      answered    = false;
      showScreen('screen-quiz');
      renderQuestion();
    }

    function renderQuestion() {
      const q     = shuffled[currentIdx];
      const total = shuffled.length;

      // Progress bar
      document.getElementById('progress-bar-fill').style.width =
        `${(currentIdx / total) * 100}%`;

      // Topic badge
      const badge = document.getElementById('topic-badge');
      badge.textContent = q.topic;
      badge.className   = 'topic-badge ' + (DOMAIN_BADGE[q.topic] || 'badge-cc');

      // Counter
      document.getElementById('progress-text').textContent =
        `Question ${currentIdx + 1} / ${total}`;

      // Question text
      document.getElementById('question-text').textContent = q.question;

      // Options
      document.getElementById('options-list').innerHTML = q.options
        .map((opt, i) => `<button class="option-btn" onclick="selectAnswer(${i})">${opt}</button>`)
        .join('');

      // Hide explanation + next button
      document.getElementById('explanation-box').classList.add('hidden');
      const nextBtn = document.getElementById('next-btn');
      nextBtn.classList.add('hidden');
      nextBtn.textContent = currentIdx === total - 1 ? 'See Results' : 'Next Question';

      answered = false;
    }

    function selectAnswer(selectedIdx) {
      if (answered) return;
      answered = true;

      const q       = shuffled[currentIdx];
      const correct = selectedIdx === q.answer;

      userAnswers.push({ topic: q.topic, correct });

      // Style the buttons
      document.querySelectorAll('.option-btn').forEach((btn, i) => {
        btn.disabled = true;
        if (i === q.answer)    btn.classList.add('correct');
        else if (i === selectedIdx) btn.classList.add('wrong');
      });

      // Show explanation
      const box     = document.getElementById('explanation-box');
      const verdict = document.getElementById('verdict-text');
      verdict.textContent  = correct ? '✓ Correct!' : '✗ Incorrect';
      verdict.className    = 'verdict ' + (correct ? 'correct-v' : 'wrong-v');
      document.getElementById('explanation-text').textContent = q.explanation;
      box.classList.remove('hidden');

      // Show next button (already has right label from renderQuestion)
      document.getElementById('next-btn').classList.remove('hidden');
    }

    function nextQuestion() {
      if (currentIdx < shuffled.length - 1) {
        currentIdx++;
        renderQuestion();
      } else {
        showResults();
      }
    }

    // ── Results Screen ─────────────────────────────────────────────
    function showResults() {
      const total   = userAnswers.length;
      const correct = userAnswers.filter(a => a.correct).length;
      const pct     = total > 0 ? correct / total : 0;

      saveRun(correct, total);

      // Score
      document.getElementById('results-score-num').textContent =
        `${correct} / ${total}`;
      document.getElementById('results-score-pct').textContent =
        `${Math.round(pct * 100)}% — ${pct >= PASS_THRESHOLD ? 'Pass' : 'Not yet'}`;

      // Banner
      const banner = document.getElementById('pass-banner');
      if (pct >= PASS_THRESHOLD) {
        banner.textContent = '✓ PASS — Great work!';
        banner.className   = 'pass-banner pass';
      } else {
        const needed = Math.round(PASS_THRESHOLD * 100);
        banner.textContent = `✗ Not yet — aim for ${needed}% to pass`;
        banner.className   = 'pass-banner fail';
      }

      // Domain breakdown
      const domains = {};
      userAnswers.forEach(({ topic, correct: c }) => {
        if (!domains[topic]) domains[topic] = { correct: 0, total: 0 };
        domains[topic].total++;
        if (c) domains[topic].correct++;
      });

      document.getElementById('domain-breakdown').innerHTML =
        Object.entries(domains).map(([topic, data]) => {
          const p     = data.total > 0 ? Math.round((data.correct / data.total) * 100) : 0;
          const color = DOMAIN_COLOR[topic] || '#0078D4';
          return `
            <div class="domain-row">
              <div class="domain-row-header">
                <span class="domain-name">${topic}</span>
                <span class="domain-score">${data.correct}/${data.total} (${p}%)</span>
              </div>
              <div class="bar-track">
                <div class="bar-fill" style="width:${p}%;background:${color}"></div>
              </div>
              <div class="domain-weight">Exam weight: ${DOMAIN_WEIGHT[topic] || 'N/A'}</div>
            </div>
          `;
        }).join('');

      showScreen('screen-results');
    }

    // ── Init ───────────────────────────────────────────────────────
    window.addEventListener('DOMContentLoaded', () => {
      initHome();
      showScreen('screen-home');
    });
  </script>
```
